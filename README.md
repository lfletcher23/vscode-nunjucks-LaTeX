# THIS IS A CUSTOM SYNTAX HIGHLIGHTER FOR A LATEX-COMPATIBLE NUNJUCKS SYNTAX:

# BASED OFF OF THE FOLLOWING NUNJUCKS SYNTAX:
tags: {
        variableStart: '#!',
        variableEnd: '!#',
        blockStart: '[##',
        blockEnd: '##]',
        commentStart: '%!',
        commentEnd: '!%'
    }

## Supports these file extensions
```
.njk, .nj, .nunjucks
```
